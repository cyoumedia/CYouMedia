/* ─────────────────────────────────────────
   MAGNETIC WRAPPER
───────────────────────────────────────── */
const Magnetic = ({ children, strength = 0.3 }) => {
  const ref = useRef(null);
  const x = useMotionValue(0);
  const y = useMotionValue(0);
  const sx = useSpring(x, { stiffness: 180, damping: 22 });
  const sy = useSpring(y, { stiffness: 180, damping: 22 });
  const onMove = (e) => {
    const r = ref.current?.getBoundingClientRect();
    if (!r) return;
    x.set((e.clientX - (r.left + r.width / 2)) * strength);
    y.set((e.clientY - (r.top + r.height / 2)) * strength);
  };
  return (
    <motion.div
      ref={ref}
      onMouseMove={onMove}
      onMouseLeave={() => {
        x.set(0);
        y.set(0);
      }}
      style={{ x: sx, y: sy }}
    >
      {children}
    </motion.div>
  );
};

/* ─────────────────────────────────────────
   MARQUEE
───────────────────────────────────────── */
const wrap = (min, max, v) => {
  const r = max - min;
  return ((((v - min) % r) + r) % r) + min;
};

const MarqueeRow = ({ items, direction = 1, speed = 0.3, className = "" }) => {
  const baseX = useMotionValue(0);
  const { scrollY } = useScroll();
  const vel = useVelocity(scrollY);
  const smooth = useSpring(vel, { stiffness: 50, damping: 30 });
  const vFac = useTransform(smooth, [0, 1000], [0, 5], { clamp: false });
  const x = useTransform(baseX, (v) => `${wrap(-50, 0, v)}%`);
  const dir = useRef(direction);
  useAnimationFrame((_, delta) => {
    let m = dir.current * speed * (delta / 1000) * 10;
    dir.current = vFac.get() < 0 ? -direction : direction;
    m += dir.current * m * vFac.get();
    baseX.set(baseX.get() + m);
  });
  const rep = [...items, ...items, ...items, ...items];
  return (
    <div className={`overflow-hidden ${className}`}>
      <motion.div style={{ x }} className="flex items-center whitespace-nowrap">
        {rep.map((item, i) => (
          <span key={i} className="flex items-center gap-5 mx-7">
            <span className="font-semibold tracking-[0.18em] uppercase">
              {item}
            </span>
            <span
              className="w-1 h-1 rounded-full opacity-25 shrink-0 inline-block"
              style={{ background: "currentColor" }}
            />
          </span>
        ))}
      </motion.div>
    </div>
  );
};
